conf t

interface ethernet 1/1-54
no shut
mtu 9216

nv overlay evpn
feature ospf
feature bgp
feature interface-vlan
feature vn-segment-vlan-based
feature nv overlay

---Underlay OSPF---

router ospf 100
  router-id 2.2.2.2

interface lo 0
  ip addr 2.2.2.2/32
  ip router ospf 100 area 0
  no shut


interface eth 1/49
  no switchport
  ip addr 192.168.2.1/30
  ip ospf network point-to-point
  ip router ospf 100 area 0
  no shut

---Control Plane MP-BGP EVPN---

router bgp 65001
  neighbor 9.9.9.9
    remote-as 65001
    update-source lo 0
    address-family l2vpn evpn
      send-community both

---


vlan 100
  vn-segment 10100

vlan 1100
  vn-segment 11000

vrf context vxlan-11000
  vni 11000
  rd auto
  address-family ipv4 unicast
    route-target both auto
    route-target both auto evpn

fabric forwarding anycast-gateway-mac 0000.2222.3333

interface vlan 100
  mtu 9216
  vrf member vxlan-11000
  ip addr 192.168.100.254/24
  fabric forwarding mode anycast-gateway
  no shut

interface vlan 300
  mtu 9216
  vrf member vxlan-11000
  ip forward
  no shut

evpn
  vni 11000 l2
  rd auto
  route-target both auto

interface lo 1
  ip address 11.11.11.11/32
  ip router ospf 100 area 0
  no shut


interface nve 1
  source-interface lo 1
  host-reachability protocol bgp
  member vni 10100
    ingress-replication protocol bgp
  member vni 11000 associate-vrf
  no shut
